import assert from "assert";
let a1 = 1;
class A6 {
    constructor(public p001: number, public p002: string) {
        p001 = p001 + 1;
        class n {
            constructor(public p001: number, public p003: string) {
                p001 = p001 + 2;
                p002 = p002 + "2";
                p003 = "4";
                this.p001 = p001 + 1;
            }
        }
        let o = new n(1, "3");
        assert(o.p001 === 4);
        assert(o.p003 === "3");
        this.p001 = p001;
        this.p002 = p002;
    }
}
let insA6 = new A6(1, "2");
assert(insA6.p001 === 2);
assert(insA6.p002 === "22");
assert(a1 === 1);
