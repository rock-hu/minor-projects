import assert from 'assert';
// Only use for testing importEqualsDeclaration in toplevel
import temp1 = require('fs');
export import temp2 = require('fs');
module X {
    export module p {
        export interface q {
            a: number;
        }
    }
    export interface p {
        b: string;
    }
}
let a: X.p.q = { a: 1 };
assert(a.a === 1);
let b: X.p = { b: "1" };
assert(b.b === "1");
module A {
    export module B {
        export class o {
            c: boolean = true;
        }
    }
}
var c: A.B.o = new A.B.o();
assert(c.c === true);
module M {
    export namespace k {
        export module l {
            export interface m {
                d: number;
            }
        }
    }
}
let d: M.k.l.m = { d: 2 };
assert(d.d === 2);
type A = number;
declare const Q1: number;
declare namespace Q2 {
    export { j };
}
let e: Q2.A = 3;
assert(e === 3);
namespace ns1 {
    namespace f {
        export var i: string = "test-importEqualsDeclaration";
    }
    import g = f.i;
    export import h = f.i;
    assert(g === "test-importEqualsDeclaration");
    assert(h === "test-importEqualsDeclaration");
}
namespace ns1 {
    assert(h === "test-importEqualsDeclaration");
}
assert(ns1.h === "test-importEqualsDeclaration");
export {};
