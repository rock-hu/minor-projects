import assert from 'assert';
let name1 = 1;
let obj = { i: name1 };
{
    const { i: h = 2 } = obj;
    h;
    assert(h === 1);
}
assert(obj.i === 1);
(function () {
    var g;
    for ({ j: g = 5 } of [{ j: 1 }]) {
        assert(g === 1);
    }
})();
(function () {
    var f;
    for ({ k: f = 5 } of [{ k: f }]) {
        assert(f === 5);
    }
})();
(function () {
    let e;
    ({ y: e = 5 } = { y: 1 });
    assert(e === 1);
})();
(function () {
    let b: string, c: {
        x: number;
    };
    let d: any = { y2: "1", l: { x: 2 } };
    ({ y2: b = '5', l: c = { x: 1 } } = d);
    assert(b === '1');
    assert(c.x === 2);
})();
(function () {
    let a;
    ({ z: a = { x: 5 } } = { z: { x: 1 } });
    assert(a.x === 1);
})();
