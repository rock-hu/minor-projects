function getDayName(a: number): string {
    let b: string = '';
    switch (a) {
        case 1:
            val1++;
            return "Monday";
        case 2:
            b = "Tuesday";
            break;
        case 3:
            return "Wednesday";
        case 4:
            return "Thursday";
        case 5:
            return "Friday";
        case 6:
            return "Saturday";
        case 7:
            return "Sunday";
        default:
            return "Invalid day";
    }
    return b;
}

assert(getDayName(1) === 'Monday');
