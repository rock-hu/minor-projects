import assert from "assert";
// 单个构造器
class A1 {
    h: number;
    constructor(m: number) {
        this.h = m;
    }
    i() {
    }
}
let insA1 = new A1(1);
assert(insA1.h === 1);
// 多个构造器
class A2 {
    j: string;
    constructor(l: 'hello');
    constructor(k: 'bye');
    constructor(j: string);
    constructor(i: any) {
        this.j = i;
    }
    k() {
    }
}
let insA2 = new A2('bee');
assert(insA2.j === 'bee');
// class声明 - 两个构造器不连续
declare namespace ns {
    class g {
        constructor();
        l: string;
        constructor(h: number);
    }
}
declare class A4 {
    constructor();
    m: string;
    constructor(a: number);
}
