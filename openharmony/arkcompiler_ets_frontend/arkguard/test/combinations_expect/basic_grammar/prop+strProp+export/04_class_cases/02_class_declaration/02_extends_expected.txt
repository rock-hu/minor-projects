import assert from "assert";
class C1 {
    t: number = 1;
    u(): string {
        return 'c1';
    }
}
class C2 extends C1 {
    v: number = 2;
}
module M1 {
    export class p {
        a1: number = 3;
        b1(): number {
            return 31;
        }
        c1(): number {
            return 32;
        }
    }
    class q extends M1.p {
        c1(): number {
            return 42;
        }
    }
    let r = new q();
    assert(r.b1() === 31);
    assert(r.c1() === 42);
}
let insC1 = new C1();
assert(insC1.u() === 'c1');
let insC2 = new C2();
assert(insC2.v === 2);
let insC3 = new M1.p();
assert(insC3.b1() === 31);
assert(insC3.c1() === 32);
