import assert from "assert";
class C1 {
    p: number = 1;
    q(): string {
        return 'c1';
    }
}
class C2 extends C1 {
    t: number = 2;
}
module M1 {
    export class C3 {
        prop_c3: number = 3;
        method_c31(): number {
            return 31;
        }
        method_c32(): number {
            return 32;
        }
    }
    class e extends M1.C3 {
        method_c32(): number {
            return 42;
        }
    }
    let f = new e();
    assert(f.method_c31() === 31);
    assert(f.method_c32() === 42);
}
let insC1 = new C1();
assert(insC1.q() === 'c1');
let insC2 = new C2();
assert(insC2.t === 2);
let insC3 = new M1.C3();
assert(insC3.method_c31() === 31);
assert(insC3.method_c32() === 32);
