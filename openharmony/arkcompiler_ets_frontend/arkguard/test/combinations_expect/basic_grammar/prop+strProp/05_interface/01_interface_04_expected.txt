import assert from 'assert';
interface Foo {
    method(a: number): string;
    g?(a: number): string;
    h: string;
    i: string;
}
class Foo {
    j: string = '';
    k(c: number): string {
        return this.method(0);
    }
}
class Bar extends Foo {
    method(b: number): string {
        return this.i;
    }
}
let a = new Bar();
assert(a.method(1) === undefined);
assert(a.g === undefined);
assert(a.h === undefined);
assert(a.i === undefined);
assert(a.j === '');
assert(a.k(1) === undefined);
