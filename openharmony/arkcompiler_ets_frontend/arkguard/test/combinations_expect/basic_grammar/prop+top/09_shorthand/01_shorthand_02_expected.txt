import assert from 'assert';
let g = 1;
let h = { g: g };
{
    const { g: p = 2 } = h;
    p;
    assert(p === 1);
}
assert(h.g === 1);
(function () {
    var o;
    for ({ i: o = 5 } of [{ i: 1 }]) {
        assert(o === 1);
    }
})();
(function () {
    var n;
    for ({ j: n = 5 } of [{ j: n }]) {
        assert(n === 5);
    }
})();
(function () {
    let m;
    ({ y: m = 5 } = { y: 1 });
    assert(m === 1);
})();
(function () {
    let j: string, k: {
        x: number;
    };
    let l: any = { y2: "1", k: { x: 2 } };
    ({ y2: j = '5', k: k = { x: 1 } } = l);
    assert(j === '1');
    assert(k.x === 2);
})();
(function () {
    let i;
    ({ z: i = { x: 5 } } = { z: { x: 1 } });
    assert(i.x === 1);
})();
