enum A {
    prop1 = 1,
    g = "2"
}
A.prop1;
A.g;
assert(A.prop1 === 1);
assert(A.g === "2");
assert(A[1] === 'prop1');

    prop1 = 1,
    g = "2"
}
B.prop1;
B.g;

assert(B.g === "2");
