import assert from 'assert';
//array
let a1 = 1;
let arr = [0, a1, 3];
assert(a1 === 1);
assert(arr[0] === 0);
assert(arr[1] === 1);
assert(arr[2] === 3);
let [b1, c1, d] = arr;
assert(b1 === 0);
assert(c1 === 1);
assert(d === 3);
[, ,] = [...arr];
//object
let obj = { a1: 1, b1: a1 + 1, c1: 3 };
assert(obj.a1 === 1);
assert(obj.b1 === 2);
assert(obj.c1 === 3);
let { c1: b2, a1: c2, b1: d2 } = obj;
assert(b2 === 3);
assert(c2 === 1);
assert(d2 === 2);
class C1 {
    a1: number;
    b1: number;
    c1: number;
    constructor([h, i, j]: Array<number>) {
        this.a1 = h;
        this.b1 = i;
        this.c1 = j;
    }
}
function f1([a, b, c]: Array<number>, { a1: e, b1: f, c1: g }: C1) {
    a;
    b;
    c;
    e;
    f;
    g;
    assert(a === 10);
    assert(b === 21);
    assert(c === 20);
    assert(e === 0);
    assert(f === 1);
    assert(g === 3);
}
f1([10, a1 + 20, 20], new C1([...arr]));
